const allAnimes = [featuredAnime, ...actionAnimes, ...romanceAnimes, ...fantasyAnimes];

export default function App() {
  const [currentPage, setCurrentPage] = useState<'home' | 'watchlater'>('home');
  const [watchLaterIds, setWatchLaterIds] = useState<number[]>([]);

  const toggleWatchLater = (id: number) => {
    setWatchLaterIds(prev => 
      prev.includes(id) 
        ? prev.filter(animeId => animeId !== id)
        : [...prev, id]
    );
  };

  const watchLaterAnimes = allAnimes.filter(anime => watchLaterIds.includes(anime.id));

  return (
    <div className="min-h-screen bg-white">
      <Header currentPage={currentPage} onNavigate={setCurrentPage} />
      
      <main className="pb-16">
        {currentPage === 'home' ? (
          <>
            <FeaturedAnime 
              {...featuredAnime} 
              isInWatchLater={watchLaterIds.includes(featuredAnime.id)}
              onToggleWatchLater={toggleWatchLater}
            />
            
            <GenreSection 
              genre="Action" 
              animes={actionAnimes} 
              watchLaterIds={watchLaterIds}
              onToggleWatchLater={toggleWatchLater}
            />
            <GenreSection 
              genre="Romance" 
              animes={romanceAnimes} 
              watchLaterIds={watchLaterIds}
              onToggleWatchLater={toggleWatchLater}
            />
            <GenreSection 
              genre="Fantasy" 
              animes={fantasyAnimes} 
              watchLaterIds={watchLaterIds}
              onToggleWatchLater={toggleWatchLater}
            />
          </>
        ) : (
          <WatchLaterPage 
            animes={watchLaterAnimes} 
            onRemove={toggleWatchLater}
          />
        )}
      </main>
    </div>
  );
}
